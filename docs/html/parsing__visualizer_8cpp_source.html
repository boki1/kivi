38  std::string stringify(const expression &expr, bool stmt) {
39  auto expr_params = expr.get_parameters();
40  switch (expr.get_type()) {
41  case expression_type::nop :
42  return "";
43  case expression_type::string :
44  return "\\\"" + expr.get_str_value() + "\\\"";
45  case expression_type::number :
46  return std::to_string(expr.get_number_val());
47  case expression_type::identifier : {
48  std::string ident_type[] = {"?", "Func", "Param", "Var", "Stmt"};
49  return ident_type[(int) expr.get_identifier().get_ident_type()] +
50  std::to_string(expr.get_identifier().get_index_within()) + "\\\"" +
51  expr.get_identifier().get_name() +
52  "\\\"";
53  }
54  case expression_type::addition :
55  return stringify_operation(expr, " + ", "()");
56  case expression_type::multiplication:
57  return stringify_operation(expr, " * ", "()");
58  case expression_type::division:
59  return stringify_operation(expr, " / ", "()");
60  case expression_type::equality :
61  return stringify_operation(expr, " == ", "()");
62  case expression_type::compare_and :
63  return stringify_operation(expr, " && ", "()");
64  case expression_type::expression_sequence :
65  return stmt ? stringify_operation(expr, "; ", "{}", true) : stringify_operation(expr, ", ", "()");
66  case expression_type::negation :
67  return "-(" + process_visualization::expect_one_param(expr) + ")";
68  case expression_type::copy :
69  return "(" + stringify(expr_params.back()) + " = " +
70  stringify(expr_params.front()) + ")";
71  case expression_type::function_call :
72  return "(" + (expr_params.empty() ? "?" :
73  stringify(expr_params.front())) + ")" + stringify_operation(expr, ", ", "()", false, 1);
74  case expression_type::compare_loop :
75  return "while " + stringify(expr_params.front()) + " " +
76  stringify_operation(expr, "; ", "{}", true, 1);
77  case expression_type::retrn :
78  return "return " + process_visualization::expect_one_param(expr);
79  }
80  return "?";

87  std::string get_next_node_token(const std::string &string, unsigned &start_idx) {
88  if (start_idx >= string.size()) {
89  return "";
90  }
91 
92  // Loop till non-space character appear
93  for (unsigned i = start_idx; i < string.size(); ++i) {
94  if (string[i] == ' ' || string[i] == ';') {
95  continue;
96  }
97  start_idx = i;
98  break;
99  }
100 
101  unsigned end_idx;
102  for (end_idx = start_idx; end_idx < string.size(); ++end_idx) {
103  if (string[end_idx] == '{' || string[end_idx] == '}') {
104  if (start_idx == end_idx) {
105  ++end_idx;
106  }
107  break;
108  }
109  if (string[end_idx] == ';') {
110  ++end_idx;
111  break;
112  }
113  }
114  return string.substr(start_idx, end_idx - start_idx);
115  }
